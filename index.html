map()

let array = ["hej","yo","tjenare"]
 
let nyarray =  array.map(    (element)=>{   // loopar och varje element ändras sedan retuneras värdena
    return `<div> <h1> ${element} </h1> </div>`
} )
            



            
filter()

let array = ["hej","yo","tjenare", "yo" ]
 
let nyarray =  array.filter(    (element)=>{   // loopar och varje element sedan retuneras värdet om det inte är "yo"
    return  element != "yo"
} )
            


            
reduce()

let price = [ 19 , 299 , 68 ]
    
let summan =  price.reduce(    (current,next)=>{   // loopar och varje element sedan retuneras värdet till nästa element
    return  current + next
} )
